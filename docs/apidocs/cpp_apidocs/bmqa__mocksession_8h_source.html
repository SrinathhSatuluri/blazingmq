1576inline bool MockSession::popPostedEvent(bmqa::MessageEvent* event)
1577{
1578 // PRECONDITIONS
1579 BSLS_ASSERT(event);
1580
1581 bslmt::LockGuard<bslmt::Mutex> guard(&d_mutex); // LOCKED
1582
1583 if (d_postedEvents.empty()) {
1584 return false; // RETURN
1585 }
1586
1587 *event = d_postedEvents.front();
1588 d_postedEvents.pop_front();
1589
1590 return true;
